ch_barony_is_valid_for_castle_trigger = {
	title_province = {
		NOT = { has_holding_type = castle_holding }
		has_ongoing_construction = no
		OR = {
			AND = {
				is_county_capital = no
				NOT = { barony.holder.top_liege = { government_has_flag = government_is_nomadic } }
			}
			AND = {
				is_county_capital = yes
				OR = {
					has_holding_type = city_holding
					has_holding_type = church_holding
					AND = {
						has_holding_type = tribal_holding
						barony.holder.top_liege = { government_has_flag = government_is_nomadic }
					}
				}
			}
		}
	}
	trigger_if = {
		limit = {
			holder = $CHARACTER$
		}
	}
	trigger_else = {
		holder.primary_title.tier = tier_barony
	}
}
